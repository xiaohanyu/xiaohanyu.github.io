题意很明确，给定一个数列 [tex]a_i(i = 0, 2, ⋯, n)[/tex]， 对于 [tex]m[/tex]，求有多少个 partial sum 能被 m 整除，所谓 partial sum 是 [tex]∑_{i = j}^k a_i, j, k ∈ 1, ⋯ n}[/tex]。

想法也很简单。第一种方法是求出所有的 partial sum，然后分别判断是否能被 m 整除，[tex]O(n^3)[/tex]的复杂度。

第二种想法是设定 [tex]σ_j = ∑_{i=0}^j a_i, j = 0, 1, 2, ⋯, n}, δ_{ij} = σ_j - σ_k = ∑_{i = k+1}^ja_i[/tex]，判断 [tex]δ_{ij}[/tex] 能否被 [tex]m[/tex] 整除。[tex]O(n^2)[/tex]复杂度。照此想法编程，确是 TLE：

#include <stdio.h>

#define max_len 10001

int main(int argc, char *argv[])
{
    int n, m;
    int inta[max_len];
    int sum[max_len+1];

    while (scanf("%d%d", &n, &m) != EOF)
    {
        int i, j;

        for (i = 0; i < n; ++i)
        {
            scanf("%d", &inta[i]);
        }

        for (i = 0; i <= n; ++i)
        {
            sum[i] = 0;
            for (j = 0; j < i; ++j)
            {
                sum[i] += inta[j];
            }
        }

        int num = 0;
        for (i = 0; i < n; ++i)
        {
            for (j = i + 1; j <= n; ++j)
            {
                if (((sum[j] - sum[i]) % m) == 0)
                {
                    num++;
                }
            }
        }
        printf ("%d\n",num);
    }

    return 0;
}

#include <string.h>
#include <stdio.h>

int c_sum[5001];

int main ()
{
    int i, n, m;
    int temp, sum, num;

    while (scanf("%d %d", &n, &m) != EOF)
    {
        sum = 0, num = 0;
        memset(c_sum, 0, sizeof(int) * (m + 1));
        for (i = 0; i < n; i++)
        {
            scanf("%d", &temp);
            sum = (sum + temp) % m;
            if (!sum)
                num++;
            c_sum[sum]++;
        }
        for (i = 0; i < m; i++)
            num += c_sum[i] * (c_sum[i] - 1) / 2;
        printf ("%d\n", num);
    }
    return 0;
}
